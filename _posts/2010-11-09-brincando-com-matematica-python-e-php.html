Estava aqui elaborando uns exercícios e me surgiu a ideia de fazer uma função para calcular as raízes de funções do 2º grau. Eu acabei implementando as funções em duas linguagens, Python e PHP:


Veja a solução em Python:


#!/usr/bin/env python
# -*- coding: utf-8 -*-

from math import sqrt

def bhaskara(a, b, c):
delta = (b * b) - (4 * a * c)
print(delta)
if delta < 0:
# Delta menor que 0, a função não tem raízes
return False
else:
delta = sqrt(delta)
r1 = (-b + delta) / (2 * a)
r2 = (-b - delta) / (2 * a)
raizes = [r1, r2]
# Retornando as raízes
return raizes

Agora em PHP:


function bhaskara($a, $b, $c) {
$delta = ($b * $b) - (4 * $a * $c);
if ($delta < 0) {
/* Delta menor que 0, a função não tem raízes */
return False;
}
else {
$delta = sqrt($delta);
$r1 = (-$b + $delta) / (2 * $a);
$r2 = (-$b - $delta) / (2 * $a);
$raizes = array(
'raiz1' => $r1,
'raiz2' => $r2
);
/* Retornando as raízes */
return $raizes;
}
}
?>
